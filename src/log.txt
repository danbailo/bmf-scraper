function ComparaDatas(dData1, dData2) {
	var iData1 = 0;
	var iData2 = 0;
	var iDia1 = new String();
	var iMes1 = new String();
	var iAno1 = new String();
	var iDia2 = new String();
	var iMes2 = new String();
	var iAno2 = new String();

	if( dData1 == "" && dData2 != "")
	{
		return 2
	}

	if( dData1 != "" && dData2 == "")
	{
		return 2
	}
	iDia1 = dData1.substring(0,2);
	iMes1 = dData1.substring(3,5);
	iAno1 = dData1.substring(6,10);
	iDia2 = dData2.substring(0,2);
	iMes2 = dData2.substring(3,5);
	iAno2 = dData2.substring(6,10);
	iData1 = parseInt(iAno1 + iMes1 + iDia1);
	iData2 = parseInt(iAno2 + iMes2 + iDia2);
	if (iData1 == iData2)
	{
		return 0
	}
	else if (iData1 < iData2)
	{
		return 1
	}
	else if (iData1 > iData2)
	{
		return 2
	}
}

function ComparaDatasMMAAAA(dData1, dData2) {
	var dDataAtual = new Date();
	var iMesAtual = new String();
	var iAnoAtual = new String();
	var iMesAtual = String((dDataAtual.getMonth())+1);
	var iAnoAtual = String(dDataAtual.getFullYear());
	
    if(iMesAtual.length == 1)
		iMesAtual = "0" + iMesAtual;
	
    var iDataAtual = 0;
	var iData1 = 0;
	var iData2 = 0;
	var iMes1 = new String();
	var iAno1 = new String();
	var iMes2 = new String();
	var iAno2 = new String();

	if( dData1 == "" && dData2 != "")
		return 2
	if( dData1 != "" && dData2 == "")
		return 2
	iMes1 = dData1.substring(0,2);
	iAno1 = dData1.substring(3,7);
	iMes2 = dData2.substring(0,2);
	iAno2 = dData2.substring(3,7);
	iData1 = parseInt(iAno1 + iMes1);
	iData2 = parseInt(iAno2 + iMes2);
	iDataAtual = parseInt(iAnoAtual + iMesAtual);
	
    if (iData1 < 190001)
		return 5; //Data1 fora do intervalo válido
	else if (iData1 == iData2 && iData1 <= iDataAtual)
		return 0; //Data1 é igual a Data2
	else if (iData1 < iData2 && iData2 <= iDataAtual)
		return 1; //Data1 é menor que Data2
	else if (iData1 > iData2)
		return 3; //Data1 é maior que Data2
	else
		return 4;
}

function Trim(sString) {
   var sStringRetorno = new String();

   sStringRetorno = sString;

   if (sStringRetorno.length > 0)
   {
      //tira espaços da esquerda
      while (sStringRetorno.charAt(0) == " ")
      {
         sStringRetorno = sStringRetorno.substring(1, sStringRetorno.length);
      }

      if (sStringRetorno.length > 0)
      {
         //tira espaços da direita
         while (sStringRetorno.charAt(sStringRetorno.length - 1) == " ")
         {
            sStringRetorno = sStringRetorno.substring(0, sStringRetorno.length - 1);
         }
      }
   }
   return (sStringRetorno);
